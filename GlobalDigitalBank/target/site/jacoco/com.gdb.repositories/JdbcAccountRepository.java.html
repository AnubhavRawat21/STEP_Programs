package com.gdb.repositories;

import com.gdb.exceptions.AccountRecordNotFoundException;
import com.gdb.exceptions.DatabaseException;
import com.gdb.models.Account;
import com.gdb.models.CurrentAccount;
import com.gdb.models.Gender;
import com.gdb.models.Privilege;
import com.gdb.models.SavingsAccount;
import com.gdb.utils.AadharUtils;

import javax.sql.DataSource;
import java.sql.*;
import java.time.LocalDate;
import java.time.LocalDateTime;

public class JdbcAccountRepository implements IAccountRepository {

    private final DataSource dataSource;

    public JdbcAccountRepository(DataSource dataSource) {
        this.dataSource = dataSource;
    }

    @Override
    public Account getAccountByAccountNumber(String accountNumber) throws AccountRecordNotFoundException {
        String sql = """
                SELECT a.id, a.account_number, a.account_type, a.name, a.balance,
                       a.pin_number, a.is_active, a.privilege, a.activated_date, a.closed_date,
                       sa.date_of_birth, sa.gender, sa.phone_number, sa.aadhar_number,
                       ca.company_name, ca.website_url, ca.registration_number
                FROM accounts a
                LEFT JOIN savings_accounts sa ON a.id = sa.account_id
                LEFT JOIN current_accounts ca ON a.id = ca.account_id
                WHERE a.account_number = ?
                """;

        try (Connection conn = dataSource.getConnection();
                PreparedStatement stmt = conn.prepareStatement(sql)) {

            stmt.setString(1, accountNumber);

            try (ResultSet rs = stmt.executeQuery()) {
                if (!rs.next()) {
                    throw new AccountRecordNotFoundException("Account not found: " + accountNumber);
                }
                return mapRowToAccount(rs);
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database error", e);
        }
    }

    @Override
    public boolean checkIfSavingsAccountExists(String aadharNumber) {
        String normalizedAadhar = AadharUtils.normalizeAadhar(aadharNumber);
        String sql = """
                SELECT COUNT(*) FROM accounts a
                JOIN savings_accounts sa ON a.id = sa.account_id
                WHERE sa.aadhar_number = ? AND a.is_active = true
                """;
        try (Connection conn = dataSource.getConnection();
                PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setString(1, normalizedAadhar);
            try (ResultSet rs = stmt.executeQuery()) {
                if (rs.next()) {
                    return rs.getInt(1) > 0;
                }
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database error", e);
        }
        return false;
    }

                """;
        try (Connection conn = dataSource.getConnection();

            stmt.setString(1, registrationNumber);

                    return rs.getInt(1) > 0;
                }
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database error", e);
        }
        return false;
    }

    }

    @Override
    public boolean checkIfCompanyIsBlacklisted(String registrationNumber) {
        if (registrationNumber == null)
            return false;
        String sql = "SELECT COUNT(*) FROM blacklist WHERE reference_type = 'REGISTRATION' AND reference_value = ?";
        try (Connection conn = dataSource.getConnection();

            try (ResultSet rs = stmt.executeQuery()) {
                if (rs.next()) {
                    return rs.getInt(1) > 0;

            }

    @Override
    public String generateAccountNumber() {
        String sql = "SELECT nextval('account_number_seq')";
        try (Connection conn = dataSource.getConnection();
                Statement stmt = conn.createStatement();
                ResultSet rs = stmt.executeQuery(sql)) {
            if (rs.next()) {
                return String.valueOf(rs.getLong(1));
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database error generating account number", e);
        }
        return null; // this should never happen
    }

    @Override
    public boolean storeAccountInformation(Account account) {
        String insertAccountSql = """
                INSERT INTO accounts (account_number, account_type, name, balance, pin_number,
                                    is_active, privilege, activated_date, closed_date)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
                RETURNING id
                """;

        String insertSavingsSql = """
                INSERT INTO savings_accounts (account_id, date_of_birth, gender, phone_number, aadhar_number)
                VALUES (?, ?, ?, ?, ?)
                """;

        String insertCurrentSql = """
                INSERT INTO current_accounts (account_id, company_name, website_url, registration_number)
                VALUES (?, ?, ?, ?)
                """;

        try (Connection conn = dataSource.getConnection()) {
            conn.setAutoCommit(false); // Start transaction

            try (PreparedStatement stmt = conn.prepareStatement(insertAccountSql)) {
                stmt.setString(1, account.getAccountNumber());
                stmt.setString(2, account instanceof SavingsAccount ? "SAVINGS" : "CURRENT");
                stmt.setString(3, account.getName());
                stmt.setDouble(4, account.getBalance());
                stmt.setString(5, account.getPinNumber());
                stmt.setBoolean(6, account.isActive());
                stmt.setString(7, account.getPrivilege().name());
                stmt.setTimestamp(8,
                        account.getActivatedDate() != null ? Timestamp.valueOf(account.getActivatedDate()) : null);
                stmt.setTimestamp(9,
                        account.getClosedDate() != null ? Timestamp.valueOf(account.getClosedDate()) : null);

                try (ResultSet rs = stmt.executeQuery()) {
                    if (!rs.next()) {
                        conn.rollback();
                        return false;
                    }
                    int accountId = rs.getInt(1);
                    account.setId(accountId); // Set the generated ID back to the model

                    if (account instanceof SavingsAccount sa) {
                        try (PreparedStatement subStmt = conn.prepareStatement(insertSavingsSql)) {
                            subStmt.setInt(1, accountId);
                            subStmt.setDate(2, Date.valueOf(sa.getDateOfBirth()));
                            subStmt.setString(3, sa.getGender().name());
                            subStmt.setString(4, sa.getPhoneNumber());
                            subStmt.setString(5, sa.getAadharNumber());
                            subStmt.executeUpdate();
                        }
                    } else if (account instanceof CurrentAccount ca) {
                        try (PreparedStatement subStmt = conn.prepareStatement(insertCurrentSql)) {
                            subStmt.setInt(1, accountId);
                            subStmt.setString(2, ca.getCompanyName());
                            subStmt.setString(3, ca.getWebsiteURL());
                            subStmt.setString(4, ca.getRegistrationNumber());
                            subStmt.executeUpdate();
                        }
                    }
                    conn.commit(); // Commit transaction
                    return true;
                }
            } catch (SQLException e) {
                conn.rollback(); // Rollback on error
                throw new DatabaseException("Database error storing account", e);
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database connection error", e);
        }
    }

    @Override
    public void closeAccount(String accountNumber, LocalDateTime closedDate) throws AccountRecordNotFoundException {
        String sql = "UPDATE accounts SET is_active = false, closed_date = ? WHERE account_number = ?";
        try (Connection conn = dataSource.getConnection();
                PreparedStatement stmt = conn.prepareStatement(sql)) {
            stmt.setTimestamp(1, Timestamp.valueOf(closedDate));
            stmt.setString(2, accountNumber);
            int rowsUpdated = stmt.executeUpdate();
            if (rowsUpdated == 0) {
                throw new AccountRecordNotFoundException("Account not found: " + accountNumber);
            }
        } catch (SQLException e) {
            throw new DatabaseException("Database error during account closure", e);
        }
    }

    private Account mapRowToAccount(ResultSet rs) throws SQLException {
        String accountType = rs.getString("account_type");

        // Common fields
        int id = rs.getInt("id");
        String accNum = rs.getString("account_number");
        String name = rs.getString("name");
        double balance = rs.getDouble("balance");
        String pin = rs.getString("pin_number");
        boolean isActive = rs.getBoolean("is_active");
        Privilege privilege = Privilege.valueOf(rs.getString("privilege"));
        LocalDateTime activatedDate = rs.getTimestamp("activated_date") != null
                ? rs.getTimestamp("activated_date").toLocalDateTime()
                : null;
        LocalDateTime closedDate = rs.getTimestamp("closed_date") != null
                ? rs.getTimestamp("closed_date").toLocalDateTime()
                : null;

        Account account;
        if ("SAVINGS".equals(accountType)) {
            SavingsAccount sa = new SavingsAccount(
                    accNum, name, balance, privilege, pin,
                    rs.getObject("date_of_birth", LocalDate.class),
                    Gender.valueOf(rs.getString("gender")),
                    rs.getString("phone_number"),
                    rs.getString("aadhar_number"));
            account = sa;
        } else { // Current account
            CurrentAccount ca = new CurrentAccount(
                    accNum, name, balance, privilege, pin,
                    rs.getString("company_name"),
                    rs.getString("website_url"),
                    rs.getString("registration_number"));
            account = ca;
        }

        account.setId(id);
        account.setActive(isActive);
        account.setActivatedDate(activatedDate);
        account.setClosedDate(closedDate);

        return account;
    }
}
